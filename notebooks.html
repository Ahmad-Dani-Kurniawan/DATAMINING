def dt1(i,j):
  P = 6
  Q = 0
  for col in ["tahun", "nama_provinsi", "nama_kabupaten_kota", "nama_kecamatan", "nama_kelurahan", "pendidikan"]:
    if data[col][i] == data[col][j]:
      Q += 1
  return (P-Q)/P

# Numerik
def dt2(i,j,h):
  result = 0
  for col in ["jumlah"]:
    result = math.pow(abs(data[col][i] - data[col][j]), h)
  return result**(1/h)
